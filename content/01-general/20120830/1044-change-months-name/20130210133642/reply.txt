```
<?php if (c::get('lang.current') != 'it'): ?>
<time datetime="<?php echo $page->date('Y-m-d') ?>"><?php echo $page->date('F j, Y') ?></time>
<?php else: ?>
<time datetime="<?php echo $page->date('Y-m-d') ?>"><?php echo strftime('%d %B %Y', $page->date()); ?></time>
<?php endif ?>
```
